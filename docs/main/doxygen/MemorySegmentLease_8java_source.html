97public final class MemorySegmentLease implements OpenableLease<MemorySegment> {
98
99 private final MemorySegmentContainer memorySegmentContainer;

101 private final Pool<OpenableLease<MemorySegment>> pool;
102
103 public MemorySegmentLease(MemorySegmentContainer bc, Pool<OpenableLease<MemorySegment>> pool) {
104 this.memorySegmentContainer = bc;
105 // initial registered parties set to 0, to be opened with open() before using
106 this.phaser = new NonTerminatingPhaser(0);
107 this.pool = pool;
108 }

110 @Override
111 public Lease<MemorySegment> sliceAt(final long offset) {
112 return new MemorySegmentSubLease(
113 new MemorySegmentContainerImpl(
114 memorySegmentContainer.id(),
115 memorySegmentContainer.memorySegment().asSlice(offset)

117 phaser
118 );
119 }

121 @Override
122 public long id() {

125
126 @Override
127 public long refs() {
128 // initial number of registered parties is 1
129 return phaser.getRegisteredParties();
130 }
131
132 @Override
133 public MemorySegment leasedObject() {
134 if (phaser.getRegisteredParties() == 0) {
135 throw new IllegalStateException(
136 "Cannot return wrapped MemorySegment, MemorySegmentLease phaser was already terminated!"
137 );
138 }
139 return memorySegmentContainer.memorySegment();
140 }
141
142 @Override
143 public boolean hasZeroRefs() {
144 return phaser.getRegisteredParties() == 0;
145 }
146
147 @Override
148 public boolean isStub() {
149 return memorySegmentContainer.isStub();
150 }
151
152 @Override
153 public void close() {
154 if (phaser.getParent() == null && phaser.getRegisteredParties() == 1) {
155 leasedObject().fill((byte) 0);
156 }
157 else {
158 throw new IllegalStateException(
159 "Cannot close lease, has <" + phaser.getRegisteredParties() + "> references."
160 );
161 }
162
163 if (phaser.arriveAndDeregister() < 0) {
164 throw new IllegalStateException("Cannot close lease, MemorySegmentLease phaser was already terminated!");
165 }
166
167 pool.offer(this);
168 }

170 @Override
171 public boolean equals(final Object o) {
172 if (o == null || getClass() != o.getClass()) {
173 return false;
174 }
175 final MemorySegmentLease that = (MemorySegmentLease) o;
176 return Objects.equals(memorySegmentContainer, that.memorySegmentContainer)
177 && Objects.equals(phaser, that.phaser) && Objects.equals(pool, that.pool);
178 }
179
180 @Override
181 public int hashCode() {
182 return Objects.hash(memorySegmentContainer, phaser, pool);
183 }
184
185 @Override
186 public void open() {
187 if (phaser.getRegisteredParties() == 0) {
188 phaser.register();
189 }
190 else {
191 throw new IllegalStateException("Cannot re-open an open lease!");
192 }
193 }
194
