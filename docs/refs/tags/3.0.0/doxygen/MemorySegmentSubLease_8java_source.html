84public final class MemorySegmentSubLease implements Lease<MemorySegment> {
85
86 private final MemorySegmentContainer memorySegmentContainer;
87 private final Phaser phaser;
88
89 public MemorySegmentSubLease(MemorySegmentContainer bc, Phaser parent) {
90 this.memorySegmentContainer = bc;
91 // initial registered parties set to 1 as subleases are not returned to the pool and can't be opened.
92 this.phaser = new NonTerminatingPhaser(parent, 1);
93 }
94
95 @Override
96 public Lease<MemorySegment> sliceAt(final long offset) {
97 if (phaser.getRegisteredParties() == 0) {
98 throw new IllegalStateException("Cannot provide slice, ref count = 0 !");
99 }
100
101 return new MemorySegmentSubLease(
102 new MemorySegmentContainerImpl(
103 memorySegmentContainer.id(),
104 memorySegmentContainer.memorySegment().asSlice(offset)
105 ),
106 phaser
107 );
108 }
109
110 @Override
111 public long id() {
112 return memorySegmentContainer.id();
113 }
114
115 @Override
116 public long refs() {
117 // initial number of registered parties is 1
118 return phaser.getRegisteredParties();
119 }
120
121 @Override
122 public MemorySegment leasedObject() {
123 if (phaser.getRegisteredParties() == 0) {
124 throw new IllegalStateException(
125 "Cannot return wrapped MemorySegment, MemorySegmentLease phaser was already terminated!"
126 );
127 }
128 return memorySegmentContainer.memorySegment();
129 }
130
131 @Override
132 public boolean hasZeroRefs() {
133 return phaser.getRegisteredParties() == 0;
134 }
135
136 @Override
137 public boolean isStub() {
138 return memorySegmentContainer.isStub();
139 }
140
141 @Override
142 public void close() {
143 if (phaser.getRegisteredParties() > 1) {
144 throw new IllegalStateException(
145 "Cannot close sub-lease, has <" + phaser.getRegisteredParties() + "> references."
146 );
147 }
148
149 if (phaser.arriveAndDeregister() < 0) {
150 throw new IllegalStateException("Cannot close lease, MemorySegmentLease phaser was already terminated!");
151 }
152 }
153
154 @Override
155 public boolean equals(final Object o) {
156 if (o == null || getClass() != o.getClass()) {
157 return false;
158 }
159 final MemorySegmentSubLease that = (MemorySegmentSubLease) o;
160 return Objects.equals(memorySegmentContainer, that.memorySegmentContainer)
161 && Objects.equals(phaser, that.phaser);
162 }
163
164 @Override
165 public int hashCode() {
166 return Objects.hash(memorySegmentContainer, phaser);
167 }
168
169 @Override
170 public String toString() {
171 return "MemorySegmentSubLease{" + "memorySegmentContainer=" + memorySegmentContainer + ", phaser=" + phaser
172 + '}';
