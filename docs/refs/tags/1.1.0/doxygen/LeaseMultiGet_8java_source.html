83public final class LeaseMultiGet implements MultiGet<OpenableLease<MemorySegment>> {
84
85 private final Pool<OpenableLease<MemorySegment>> leasePool;
86
87 public LeaseMultiGet(final Pool<OpenableLease<MemorySegment>> leasePool) {
88 this.leasePool = leasePool;
89 }
90
91 @Override
92 public List<OpenableLease<MemorySegment>> get(final long bytesCount) {
93 long currentSize = 0;
94 final List<OpenableLease<MemorySegment>> leases = new ArrayList<>();
95 while (currentSize < bytesCount) {
96 final OpenableLease<MemorySegment> lease = leasePool.get();
97 if (lease.isStub()) {
98 throw new IllegalStateException("Pool supplied a stub lease!");
99 }
100 leases.add(lease);
101 currentSize += lease.leasedObject().byteSize();
102 }
103 return leases;
104 }
105
106 @Override
107 public boolean equals(final Object o) {
108 if (o == null || getClass() != o.getClass()) {
109 return false;
110 }
111 final LeaseMultiGet that = (LeaseMultiGet) o;
112 return Objects.equals(leasePool, that.leasePool);
113 }
114
115 @Override
116 public int hashCode() {
117 return Objects.hashCode(leasePool);
118 }
119}
