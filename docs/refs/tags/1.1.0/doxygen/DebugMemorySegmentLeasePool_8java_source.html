101 private static final Logger LOGGER = LoggerFactory.getLogger(DebugMemorySegmentLeasePool.class);

116 public DebugMemorySegmentLeasePool() {
117 this.segmentSize = 4096;
118 this.queue = new ConcurrentLinkedQueue<>();
119 this.leaseStub = new MemorySegmentLeaseStub();
120 this.close = new AtomicBoolean();
121 this.bufferId = new AtomicLong();
122 this.lock = new ReentrantLock();
123 }
124
125 public OpenableLease<MemorySegment> get() {
126 if (close.get()) {
127 return new MemorySegmentLeaseStub();
128 }
129
130 // always get new lease
131 final ArenaMemorySegmentLeaseSupplier supplier = new ArenaMemorySegmentLeaseSupplier(
132 Arena.ofShared(),
133 segmentSize,
134 bufferId.get()
135 );
136 final OpenableLease<MemorySegment> lease = supplier.apply(this);
137 suppliers.put(bufferId.incrementAndGet(), supplier);
138
139 if (LOGGER.isDebugEnabled()) {
140 LOGGER.debug("returning bufferLease id <{}> with refs <{}>", lease.id(), lease.refs());
141 }
142
143 return lease;
144 }
145
151 @Override
152 public void offer(OpenableLease<MemorySegment> object) {
153 // debug pool, instead of returning to pool arena is closed and memorySegment is discarded.
154 if (!object.isStub()) {
155 suppliers.get(object.id()).accept(object);
156 suppliers.get(object.id()).close();
157 }
158 }
159
164 public void close() {
165 LOGGER.debug("close called");
166 close.set(true);
167 }

169 @Override
170 public boolean equals(final Object o) {
171 if (o == null || getClass() != o.getClass()) {
172 return false;
173 }
174 final DebugMemorySegmentLeasePool that = (DebugMemorySegmentLeasePool) o;

176 .equals(queue, that.queue) && Objects.equals(leaseStub, that.leaseStub) && Objects
177 .equals(close, that.close)
178 && Objects.equals(bufferId, that.bufferId) && Objects.equals(lock, that.lock);
179 }
180
