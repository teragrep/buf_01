150 @Override
151 public boolean hasNext() {
152 final boolean rv;
153 if (limit == -1) {
154 // limit not set, ignore
155 rv = currentOffset < origin.leasedObject().byteSize();
156 }
157 else {
158 rv = currentOffset < Math.min(limit, origin.leasedObject().byteSize());
159 }
160 return rv;
161 }

163 @Override
164 public byte next() {
165 if (!hasNext()) {
166 throw new IndexOutOfBoundsException("Reached end of segment or limit, cannot provide next byte");
167 }
168
169 return origin.leasedObject().get(ValueLayout.JAVA_BYTE, currentOffset++);
170 }

174 if (!hasNext()) {
175 throw new IndexOutOfBoundsException("Reached end of segment or limit, cannot write to next byte");
176 }
177
178 origin.leasedObject().set(ValueLayout.JAVA_BYTE, currentOffset++, b);
179 }

187 public void position(final long newPosition) {
188 final long segmentByteSize = leasedObject().byteSize();
189 if (newPosition < 0 || newPosition > segmentByteSize) {
190 throw new IndexOutOfBoundsException(
191 "New position was out of bounds; expected value between 0 and " + segmentByteSize + "; was "
192 + newPosition
193 );
194 }
195
196 final long currentLimit = limit;
197 if (newPosition > currentLimit && currentLimit != -1) {
198 throw new IndexOutOfBoundsException(
199 "New position was larger than the limit, limit=" + limit + " newPosition=" + newPosition
200 );
201 }
202
203 currentOffset = newPosition;
204 }

213 final long segmentByteSize = leasedObject().byteSize();
214 if (newLimit < -1 || newLimit > segmentByteSize) {
215 throw new IndexOutOfBoundsException(
216 "New limit was out of bounds; expected value between -1 and " + segmentByteSize + "; was "
217 + newLimit
218 );
219 }
220
221 limit = newLimit;
222
223 if (newLimit < currentPosition() && newLimit != -1) {
224 position(newLimit);
225 }
226 }
